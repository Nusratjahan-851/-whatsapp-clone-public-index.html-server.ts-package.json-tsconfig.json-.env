const handleSafeDownload = (fileUrl: string, fileName: string = "download") => {
  try {
    if (fileUrl.startsWith("data:")) {
      const parts = fileUrl.split(";base64,");
      if (parts.length === 2) {
        const contentType = parts[0].split(":")[1];
        const raw = window.atob(parts[1]);
        const rawLength = raw.length;
        const uInt8Array = new Uint8Array(rawLength);
        for (let i = 0; i < rawLength; ++i) {
          uInt8Array[i] = raw.charCodeAt(i);
        }
        const blob = new Blob([uInt8Array], { type: contentType });
        const blobUrl = URL.createObjectURL(blob);
        
        const link = document.createElement("a");
        link.href = blobUrl;
        link.download = fileName;
        link.target = "_blank";
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        
        setTimeout(() => {
          URL.revokeObjectURL(blobUrl);
        }, 100);
        return;
      }
    }
    
    // Fallback for non-base64 URLs
    const link = document.createElement("a");
    link.href = fileUrl;
    link.download = fileName;
    link.target = "_blank";
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  } catch (error) {
    console.error("Failed to download file safely:", error);
    window.open(fileUrl, "_blank");
  }
};

  isDark: boolean;
  contacts: Contact[];

    type: "text" | "image" | "voice" | "file" | "video" | "sticker" | "contact" | "location" | "poll",

  onTranscribe: (messageId: string) => void;
  onOpenSearch: () => void;
  scrollToMessageId: string | null;
  clearScrollToMessage: () => void;

  // Custom callback links
  onSetQuotedMessage: (msg: Message) => void;
  onUpdatePollVote: (msgId: string, optionId: string) => void;
  onAddReaction: (msgId: string, emoji: string) => void;
  onUpdateContactSettings: (contactId: string, updates: Partial<Contact>) => void;
  onSelectContactById: (id: string | null) => void;
  onReloadHistory: (contactId: string, count?: number) => void;
  onViewProfile?: (contactId: string) => void;

  // Message Actions
  onDeleteMessage: (msgId: string) => void;
  onPinMessage: (msgId: string) => void;
  onStarMessage: (msgId: string) => void;
  onForwardMessage: (msg: Message, targetContactId: string) => void;

  messages,
  isDark,
  contacts,

  onTranscribe,
  onOpenSearch,

  clearScrollToMessage,
  onSetQuotedMessage,

  onAddReaction,

  onReloadHistory,
  onViewProfile,
  onDeleteMessage,

  onStarMessage,
  onForwardMessage,
}) => {

  const [activeZoomImage, setActiveZoomImage] = useState<string | null>(null);
  const [hoveredMessageId, setHoveredMessageId] = useState<string | null>(null);
  const [showGroupDrawer, setShowGroupDrawer] = useState(false);
  const [newMemberName, setNewMemberName] = useState("");

  const [loadingOlder, setLoadingOlder] = useState(false);
  const [isReloading, setIsReloading] = useState(false);

  const messageRefs = useRef<Record<string, HTMLDivElement | null>>({});

  const [downloadedMessages, setDownloadedMessages] = useState<Record<string, boolean>>({});
  const [downloadProgress, setDownloadProgress] = useState<Record<string, number>>({});

  // Long press / Context menu / Star / Pin / Forward state
  const [contextMenuMsg, setContextMenuMsg] = useState<{ msg: Message; x: number; y: number } | null>(null);
  const [longPressTimer, setLongPressTimer] = useState<NodeJS.Timeout | null>(null);
  const [isLongPressActive, setIsLongPressActive] = useState(false);
  const [isForwardModalOpen, setIsForwardModalOpen] = useState(false);
  const [forwardMessageTarget, setForwardMessageTarget] = useState<Message | null>(null);
  const [forwardSearchQuery, setForwardSearchQuery] = useState("");
  const [toastMessage, setToastMessage] = useState<string | null>(null);
  const [showSubEmojiPicker, setShowSubEmojiPicker] = useState(false);
  const [forwardedStatus, setForwardedStatus] = useState<Record<string, boolean>>({});

  const showToast = (text: string) => {
    setToastMessage(text);
    setTimeout(() => {
      setToastMessage(null);
    }, 2500);
  };

  const handlePressStart = (e: React.MouseEvent | React.TouchEvent, msg: Message) => {
    // Only trigger on primary click (button 0) if mouse
    if (e.type === "mousedown" && (e as React.MouseEvent).button !== 0) return;
    
    if (longPressTimer) clearTimeout(longPressTimer);
    setIsLongPressActive(false);

    // Get position coordinates for overlay placement
    const clientX = "touches" in e ? e.touches[0].clientX : (e as React.MouseEvent).clientX;
    const clientY = "touches" in e ? e.touches[0].clientY : (e as React.MouseEvent).clientY;

    const timer = setTimeout(() => {
      setIsLongPressActive(true);
      if (navigator.vibrate) navigator.vibrate(50);
      setContextMenuMsg({ msg, x: clientX, y: clientY });
    }, 550); // standard long-press duration (550ms)
    
    setLongPressTimer(timer);
  };

  const handlePressEnd = () => {
    if (longPressTimer) {
      clearTimeout(longPressTimer);
      setLongPressTimer(null);
    }
  };

  const handleContextMenu = (e: React.MouseEvent, msg: Message) => {
    e.preventDefault();
    if (longPressTimer) clearTimeout(longPressTimer);
    setContextMenuMsg({ msg, x: e.clientX, y: e.clientY });
  };

  useEffect(() => {

      const list = await loadSetting<string[]>("wa_downloaded_messages", []);
      const record: Record<string, boolean> = {};
      list.forEach((id) => {
        record[id] = true;
      });
      setDownloadedMessages(record);
    }
    fetchDownloaded();
  }, []);

  const startDownload = (msgId: string) => {
    if (downloadProgress[msgId] !== undefined) return; // already downloading
    
    // Start from 0
    setDownloadProgress((prev) => ({ ...prev, [msgId]: 0 }));
    
    const totalDuration = 1200 + Math.random() * 800; // 1.2s to 2.0s duration
    const steps = 10;
    const intervalTime = totalDuration / steps;
    let currentStep = 0;
    
    const interval = setInterval(() => {
      currentStep++;
      const progress = Math.min(Math.round((currentStep / steps) * 100), 100);
      
      setDownloadProgress((prev) => ({ ...prev, [msgId]: progress }));
      
      if (progress >= 100) {
        clearInterval(interval);
        // Save downloaded message id
        setDownloadedMessages((prev) => {
          const next = { ...prev, [msgId]: true };
          // Save to IndexedDB settings
          const list = Object.keys(next);
          saveSetting("wa_downloaded_messages", list);
          return next;
        });
        // Clean download progress state
        setDownloadProgress((prev) => {
          const next = { ...prev };
          delete next[msgId];
          return next;
        });
      }
    }, intervalTime);

    if (explicitSize) return explicitSize;
    let hash = 0;
    for (let i = 0; i < msgId.length; i++) {
      hash = msgId.charCodeAt(i) + ((hash << 5) - hash);
    }
    hash = Math.abs(hash);
    if (type === "voice") {
      const kb = (hash % 150) + 20;
      return `${kb} kB`;
    }
    if (type === "sticker") {
      const kb = (hash % 60) + 15;
      return `${kb} kB`;
    }
    if (type === "image") {
      const mb = ((hash % 180) + 40) / 100;
      return `${mb.toFixed(1)} MB`;
    }
    if (type === "video") {
      const mb = ((hash % 850) + 150) / 100;
      return `${mb.toFixed(1)} MB`;
    }
    const mb = ((hash % 450) + 50) / 100;
    return `${mb.toFixed(1)} MB`;
  };

    msg,

    className = "",

    msg: Message;
    children: React.ReactNode;

  // Robust, anchor-aware scroll-management system
  useEffect(() => {

    if (!container) return;

    const isContactChanged = contact.id !== lastContactIdRef.current;
    
    if (isContactChanged) {
      // Switch contact: Scroll to bottom immediately
      container.scrollTop = container.scrollHeight;
      lastContactIdRef.current = contact.id;
      lastMessagesLengthRef.current = messages.length;
      firstMessageIdRef.current = messages[0]?.id || null;
      return;
    }

    // Same contact: messages modified
    const currentFirstMsgId = messages[0]?.id || null;
    const previousFirstMsgId = firstMessageIdRef.current;
    
    if (currentFirstMsgId !== previousFirstMsgId && previousFirstMsgId !== null) {

      const scrollDiff = newScrollHeight - previousScrollHeightRef.current;
      container.scrollTop = previousScrollTopRef.current + scrollDiff;
    } else {

      const isSentByMe = lastMsg?.sender === "me";

      if (isSentByMe || isNearBottom) {
        container.scrollTop = container.scrollHeight;
      }

    lastMessagesLengthRef.current = messages.length;
    firstMessageIdRef.current = currentFirstMsgId;

  const handleScroll = () => {

        setTimeout(() => {
          element.classList.remove("bg-yellow-400/30", "dark:bg-yellow-500/20");
          clearScrollToMessage();

      }

  };

  };

    const updated = (contact.participants || []).map((p) =>

    if (contact.inviteLink) {

    return { totalVotes, optionsWithPercentages };

  return (
    <div className="flex-1 flex min-h-0 overflow-hidden relative">
      {/* Primary Conversation Screen */}
      <div className="flex-1 flex flex-col min-h-0 overflow-hidden relative">
        {/* Header Panel */}

                    onMouseDown={(e) => handlePressStart(e, msg)}

                    className={`max-w-[85%] md:max-w-[70%] lg:max-w-[55%] rounded-lg px-3 py-2 shadow-xs relative text-sm flex flex-col gap-1 border border-black/5 cursor-pointer select-none ${
                      isMe

                    {msg.type === "sticker" && msg.fileUrl && (
                      <MediaBlurWrapper msg={msg}>
                        <div className="flex justify-center p-1 select-none">
                          <img
                            src={msg.fileUrl || null}
                            alt="sticker"
                            className="w-24 h-24 object-contain animate-bounce"
                            referrerPolicy="no-referrer"

                        </div>
                      </MediaBlurWrapper>
                    )}

                    {/* TYPE 2: FILE / DOCUMENT */}
                    {msg.type === "file" && (
                      <MediaBlurWrapper msg={msg}>
                        <div className="flex items-center gap-3 p-2 bg-black/5 dark:bg-white/5 rounded-md border border-black/10 dark:border-white/10 mb-1 max-w-full sm:max-w-[320px] md:max-w-[400px]">

                          {msg.fileUrl && (
                            <button
                              onClick={(e) => {
                                e.preventDefault();
                                handleSafeDownload(msg.fileUrl, msg.fileName);

                              className="p-1.5 rounded-full hover:bg-black/10 dark:hover:bg-white/10 text-inherit shrink-0 cursor-pointer"

                            >
                              <Download size={14} />
                            </button>

                        </div>
                      </MediaBlurWrapper>
                    )}

                    {/* TYPE 3: IMAGE */}
                    {msg.type === "image" && msg.fileUrl && (
                      <MediaBlurWrapper msg={msg}>
                        <div className="relative group rounded-md overflow-hidden mb-1 border border-black/10 dark:border-white/10 select-none max-w-full sm:max-w-[320px] md:max-w-[400px]">
                          <img
                            src={msg.fileUrl || null}
                            alt="Shared File"
                            className="max-h-64 object-cover w-full cursor-pointer hover:brightness-95 transition-all"
                            onClick={() => setActiveZoomImage(msg.fileUrl || null)}
                            referrerPolicy="no-referrer"
                          />
                          <button
                            onClick={() => setActiveZoomImage(msg.fileUrl || null)}
                            className="absolute bottom-2 right-2 bg-black/60 p-1.5 rounded-full text-white opacity-0 group-hover:opacity-100 transition-opacity cursor-pointer"
                            title="Zoom"
                          >
                            <ZoomIn size={14} />
                          </button>
                        </div>
                      </MediaBlurWrapper>
                    )}

                    {/* TYPE 4: VIDEO */}
                    {msg.type === "video" && msg.fileUrl && (
                      <MediaBlurWrapper msg={msg}>
                        <div className="relative rounded-md overflow-hidden mb-1 border border-black/10 dark:border-white/10 select-none bg-black max-w-full sm:max-w-[320px] md:max-w-[400px]">
                          <video
                            src={msg.fileUrl}
                            controls
                            className="max-h-64 w-full rounded"
                          />
                        </div>
                      </MediaBlurWrapper>
                    )}

                    {/* TYPE 5: VOICE */}
                    {msg.type === "voice" && (
                      <MediaBlurWrapper msg={msg}>
                        <AudioPlayer message={msg} isDark={isDark} onTranscribe={onTranscribe} />
                      </MediaBlurWrapper>
                    )}

                    {/* TYPE 6: LOCATION */}
                    {msg.type === "location" && msg.locationData && (
                      <div className="flex flex-col gap-1.5 mb-1 p-2 rounded bg-black/10 dark:bg-white/5 border border-black/10 dark:border-white/10">
                        <div className="flex items-center gap-2 text-red-500 font-bold text-xs">
                          <MapPin size={16} />
                          <span>Location Shared</span>
                        </div>
                        <div className="text-xs">
                          <h4 className="font-bold">{msg.locationData.name}</h4>
                          <p className="opacity-70 text-[10px] mt-0.5">{msg.locationData.address}</p>

                        <div className="text-[10px] font-mono opacity-50">
                          Lat: {msg.locationData.latitude.toFixed(4)}, Lng: {msg.locationData.longitude.toFixed(4)}
                        </div>
                        <a
                          href={`https://www.google.com/maps/search/?api=1&query=${msg.locationData.latitude},${msg.locationData.longitude}`}

                          rel="noopener noreferrer"
                          className="mt-2 py-1.5 bg-[#00a884] hover:bg-[#008069] text-white font-bold text-center text-xs rounded flex items-center justify-center gap-1 shadow-sm cursor-pointer"
                        >
                          <span>Open in Google Maps</span>
                          <ExternalLink size={10} />
                        </a>
                      </div>
                    )}

                    {/* TYPE 7: CONTACT CARD */}
                    {msg.type === "contact" && msg.contactCard && (
                      <div className="flex flex-col gap-2 p-2.5 rounded bg-black/10 dark:bg-white/5 border border-black/10 dark:border-white/10 min-w-[200px]">
                        <div className="flex items-center gap-2.5">
                          <div className="w-10 h-10 rounded-full bg-gray-200 border border-black/10 overflow-hidden flex items-center justify-center">
                            {msg.contactCard.avatar ? (
                              <img src={msg.contactCard.avatar || null} alt="Shared Contact" className="w-full h-full object-cover" referrerPolicy="no-referrer" />
                            ) : (
                              <User size={16} className="text-gray-500" />
                            )}
                          </div>
                          <div className="min-w-0">
                            <h4 className="font-bold text-xs truncate">{msg.contactCard.name}</h4>
                            <p className="text-[10px] opacity-60 truncate">{msg.contactCard.phone}</p>
                          </div>
                        </div>
                        <p className="text-[10px] italic opacity-70 border-t border-black/10 dark:border-white/10 pt-1.5 truncate">
                          {msg.contactCard.about || "Available"}
                        </p>
                        <button
                          onClick={() => {
                            // Find matching contact or simulate a redirect
                            alert(`Redirecting to conversation with ${msg.contactCard?.name}...`);
                          }}
                          className="w-full mt-1.5 py-1.5 border border-[#00a884] hover:bg-[#00a884]/15 text-[#00a884] font-bold text-xs rounded transition-colors cursor-pointer text-center"
                        >
                          Message
                        </button>
                      </div>
                    )}

                    {/* TYPE 8: POLLS (ACTIVE INTERACTIVE VOTING!) */}
                    {msg.type === "poll" && msg.pollData && (() => {
                      const { totalVotes, optionsWithPercentages } = getPollResults(msg.pollData);
                      return (
                        <div className="flex flex-col gap-2 p-2 bg-black/10 dark:bg-white/5 rounded border border-black/15 dark:border-white/15 min-w-[240px] max-w-full sm:max-w-[320px] md:max-w-[400px]">
                          <div className="font-bold text-xs border-b pb-1.5 border-black/10 dark:border-white/10">
                            📊 {msg.pollData.question}
                          </div>
                          <div className="space-y-2 mt-1">
                            {optionsWithPercentages.map((opt) => {
                              const didIVote = opt.votes.includes("You (Owner)");
                              return (
                                <button
                                  key={opt.id}
                                  onClick={() => onUpdatePollVote(msg.id, opt.id)}
                                  className={`w-full text-left p-2 rounded relative overflow-hidden flex flex-col gap-0.5 border text-xs transition-all cursor-pointer ${
                                    didIVote
                                      ? "border-emerald-500/40 bg-emerald-500/10"
                                      : "border-black/5 hover:bg-black/5 dark:hover:bg-white/5"
                                  }`}
                                >
                                  {/* Percentage background filler */}
                                  <div
                                    className="absolute left-0 top-0 bottom-0 bg-emerald-500/15 transition-all duration-500 ease-out z-0"
                                    style={{ width: `${opt.percentage}%` }}
                                  />
                                  <div className="flex justify-between items-center relative z-10 w-full font-semibold">
                                    <span>{opt.text}</span>
                                    <span>{opt.votes.length} votes</span>
                                  </div>
                                  <div className="flex items-center justify-between text-[9px] opacity-60 relative z-10 mt-0.5">
                                    <span className="truncate max-w-[150px]">
                                      {opt.votes.length > 0 ? opt.votes.join(", ") : "No votes"}
                                    </span>
                                    <span>{opt.percentage}%</span>
                                  </div>
                                </button>
                              );
                            })}
                          </div>
                          <div className="text-[9px] text-center opacity-55 border-t border-black/5 pt-1.5">
                            Total: {totalVotes} votes • Click choice to vote
                          </div>
                        </div>
                      );
                    })()}

                    {/* GENERAL TEXT DESCRIPTION FOR BASIC MESSAGE CHATS */}
                    {msg.type === "text" && (
                      <p className="whitespace-pre-wrap break-words leading-relaxed text-sm pr-12">
                        {msg.text}
                      </p>
                    )}

                    {/* Reaction Pill Indicator Overlay */}
                    {msg.reactions && msg.reactions.length > 0 && (
                      <div className="absolute -bottom-2.5 right-2 flex items-center gap-0.5 bg-white dark:bg-[#222e35] px-1.5 py-0.5 rounded-full shadow border border-black/10 text-xs select-none z-10 scale-95">
                        {Array.from(new Set(msg.reactions.map((r) => r.emoji))).slice(0, 3).map((emoji, idx) => (
                          <span key={idx}>{emoji}</span>
                        ))}
                        <span className="text-[9px] font-bold opacity-60 pl-0.5">{msg.reactions.length}</span>
                      </div>
                    )}

                    {/* Metadata indicators */}
                    <div className="absolute bottom-1 right-2 flex items-center gap-1.5 text-[9px] opacity-50 select-none pointer-events-none">
                      {msg.isStarred && <Star size={9} className="fill-amber-400 stroke-amber-400" />}
                      {msg.isPinned && <Pin size={9} className="rotate-45 text-emerald-500 fill-emerald-500" />}
                      <span>{msg.timestamp}</span>
                      {isMe && renderStatusTicks(msg.status)}
                    </div>
                  </div>

                  {/* Reply and reaction hover shortcuts panel */}
                  {isHovered && (
                    <div
                      className={`absolute top-1/2 -translate-y-1/2 flex items-center gap-1 p-1 bg-white dark:bg-[#202c33] rounded-full shadow-lg border border-black/10 z-20 ${
                        isMe ? "right-full mr-2" : "left-full ml-2"
                      }`}
                      style={{ animation: "scaleUp 0.1s ease-out" }}
                    >
                      {/* 1. Emoji picker popup row */}
                      {["👍", "❤️", "😂", "😮", "😢", "🙏"].map((emoji) => (
                        <button
                          key={emoji}
                          onClick={() => onAddReaction(msg.id, emoji)}
                          className="hover:scale-125 transition-transform text-sm px-1 cursor-pointer"
                        >
                          {emoji}
                        </button>
                      ))}

                      {/* 2. Quoted Reply action */}
                      <button
                        onClick={() => onSetQuotedMessage(msg)}
                        className="p-1 hover:bg-black/5 dark:hover:bg-white/10 rounded-full text-emerald-500 cursor-pointer"
                        title="Quote and Reply"
                      >
                        <CornerUpLeft size={14} />
                      </button>
                    </div>
                  )}
                </div>
              );
            })}
          </>
          )}
        </div>

        {/* Scroll Down Floating Indicator */}
        {showScrollDown && (
          <button
            onClick={scrollToBottom}
            className={`absolute bottom-16 right-6 p-2.5 rounded-full shadow-lg border transition-all z-20 cursor-pointer hover:scale-105 ${
              isDark
                ? "bg-[#202c33] border-gray-700 text-[#00a884] hover:bg-[#2a3942]"
                : "bg-white border-gray-200 text-[#00a884] hover:bg-gray-100"
            }`}
          >
            <ArrowDown size={18} />
          </button>
        )}
      </div>

      {/* COLLAPSIBLE SIDEBAR: GROUP DETAILS / CONTACT INFO DRAWER */}
      {showGroupDrawer && (
        <div
          className={`w-[320px] md:w-[350px] h-full border-l flex flex-col shrink-0 select-none z-30 relative ${
            isDark ? "bg-[#111b21] border-gray-800 text-white" : "bg-white border-gray-200 text-gray-800"
          }`}
          style={{ animation: "slideLeft 0.15s ease-out" }}
        >
          {/* Drawer Header */}
          <div
            className={`flex items-center justify-between px-4 py-3 border-b shrink-0 ${
              isDark ? "bg-[#202c33] border-gray-800" : "bg-[#f0f2f5] border-gray-200"
            }`}
          >
            <h3 className="font-bold text-sm">
              {contact.isGroup ? "Group Details" : "Contact Information"}
            </h3>
            <button
              onClick={() => setShowGroupDrawer(false)}
              className="p-1 rounded-full hover:bg-black/10 dark:hover:bg-white/10 cursor-pointer"
            >
              <X size={18} />
            </button>
          </div>

          {/* Drawer Content */}
          <div className="flex-1 overflow-y-auto overflow-x-hidden p-4 space-y-5">
            {/* 1. Large Avatar Area */}
            <div className="flex flex-col items-center text-center pb-4 border-b border-black/5 dark:border-white/5">
              <div
                className={`w-24 h-24 rounded-full flex items-center justify-center text-3xl font-bold mb-3 overflow-hidden ${
                  contact.id === "meta-ai"
                    ? "bg-gradient-to-tr from-purple-500 via-pink-500 to-blue-500"
                    : "bg-emerald-500/10 text-emerald-500 border border-emerald-500/20"
                }`}
              >
                {contact.id === "meta-ai" ? (
                  "M"
                ) : contact.avatar && contact.avatar.startsWith("http") ? (
                  <img src={contact.avatar || null} alt={contact.name} className="w-full h-full object-cover" referrerPolicy="no-referrer" />

                  (contact.name || "User").slice(0, 2).toUpperCase()
                )}
              </div>
              <h2 className="font-bold text-base leading-tight flex items-center gap-1 justify-center">
                {contact.name}
                {contact.verified && <span className="text-[#53bdeb] text-sm">✔️</span>}
              </h2>
              <p className="text-xs opacity-60 mt-1">
                {contact.isGroup ? "WhatsApp Group Chat" : contact.phoneNumber || "No Phone Number"}
              </p>
            </div>

            {/* 2. Group Description / Contact About */}
            <div className="space-y-1">
              <h4 className="text-[10px] font-bold uppercase opacity-55 tracking-wider">
                {contact.isGroup ? "Group Description" : "About & Phone Status"}
              </h4>
              {contact.isGroup ? (
                <div className="space-y-2">
                  <textarea
                    value={contact.groupDescription || ""}
                    onChange={(e) => onUpdateContactSettings(contact.id, { groupDescription: e.target.value })}
                    placeholder="Enter group description or announcement..."
                    rows={3}
                    className={`w-full p-2 text-xs rounded border outline-none focus:ring-1 focus:ring-[#00a884] ${
                      isDark ? "bg-[#2a3942] border-gray-700" : "bg-gray-50 border-gray-200"
                    }`}
                  />
                </div>
              ) : (
                <p className="text-xs italic p-2 rounded bg-black/5 dark:bg-white/5">
                  {contact.about || "No status set."}
                </p>
              )}
            </div>

            {/* 3. Visual Media Gallery Preview (Shared images, videos, documents) */}
            <div className="space-y-2 pt-3 border-t border-black/5 dark:border-white/5">
              <h4 className="text-[10px] font-extrabold uppercase opacity-60 tracking-wider flex justify-between items-center text-[#00a884] dark:text-[#00e676]">
                <span>Shared Media & Docs</span>
                <span className="text-[9px] lowercase font-normal opacity-70">
                  {messages.filter((m) => ["image", "video", "file"].includes(m.type)).length} files
                </span>
              </h4>
              
              {messages.filter((m) => ["image", "video", "file"].includes(m.type)).length === 0 ? (
                <p className="text-[10px] opacity-50 italic px-1">No media files shared in this conversation yet.</p>
              ) : (
                <div className="grid grid-cols-4 gap-2">
                  {messages
                    .filter((m) => ["image", "video", "file"].includes(m.type))
                    .slice(0, 8)
                    .map((m) => {
                      if (m.type === "image") {
                        return (
                          <div
                            key={m.id}
                            onClick={() => setActiveZoomImage(m.fileUrl || m.text)}
                            className="aspect-square rounded-lg bg-black/10 overflow-hidden cursor-pointer hover:opacity-80 transition-opacity border border-black/5 dark:border-white/5 relative group shrink-0"

                          >
                            <img src={m.fileUrl || m.text} alt="preview" className="w-full h-full object-cover" referrerPolicy="no-referrer" />
                          </div>

                      } else if (m.type === "video") {
                        return (
                          <div
                            key={m.id}
                            className="aspect-square rounded-lg bg-black flex items-center justify-center text-white cursor-pointer relative overflow-hidden border border-black/5 dark:border-white/5 shrink-0"
                            onClick={() => setActiveZoomImage(m.fileUrl || m.text)}
                            title="Play video"
                          >
                            <Video size={14} className="text-[#00a884] dark:text-[#00e676]" />
                            <div className="absolute inset-0 bg-black/10 hover:bg-black/0 transition-colors" />
                          </div>
                        );
                      } else {
                        return (
                          <div
                            key={m.id}
                            className="aspect-square rounded-lg bg-black/5 dark:bg-white/5 flex flex-col items-center justify-center p-1 text-center border border-black/5 dark:border-white/5 shrink-0 relative cursor-pointer hover:bg-black/10"
                            title={m.fileName || "document"}
                            onClick={() => {
                              if (m.fileUrl) {
                                handleSafeDownload(m.fileUrl, m.fileName || "document");
                              }
                            }}
                          >
                            <FileText size={14} className="text-emerald-500" />
                            <span className="text-[7px] truncate w-full mt-1 opacity-80 font-mono leading-none">{m.fileName || "doc"}</span>
                          </div>
                        );
                      }
                    })}
                </div>
              )}
            </div>

            {/* 3. Group Administration settings & Participant control */}
            {contact.isGroup && (
              <div className="space-y-3.5 pt-2 border-t border-black/5 dark:border-white/5">
                <div className="flex justify-between items-center">
                  <h4 className="text-[10px] font-bold uppercase opacity-55 tracking-wider">
                    Participants ({contact.participants?.length || 0})
                  </h4>
                  {contact.inviteLink && (
                    <button
                      onClick={handleCopyInviteLink}
                      className="text-[#00a884] hover:underline text-[10px] font-bold flex items-center gap-0.5 cursor-pointer"
                    >
                      <Copy size={10} /> Copy Link
                    </button>
                  )}
                </div>

                {/* Add participant input form */}
                <form onSubmit={handleAddMember} className="flex gap-2">
                  <input
                    type="text"
                    placeholder="Simulate new user..."
                    value={newMemberName}
                    onChange={(e) => setNewMemberName(e.target.value)}
                    className={`flex-1 px-2.5 py-1.5 text-xs rounded border outline-none focus:ring-1 focus:ring-[#00a884] ${
                      isDark ? "bg-[#2a3942] border-gray-700" : "bg-gray-50 border-gray-200"
                    }`}
                  />
                  <button
                    type="submit"
                    className="p-2 bg-emerald-500 hover:bg-emerald-600 text-white rounded shrink-0 cursor-pointer"
                    title="Add Member"
                  >
                    <UserPlus size={14} />
                  </button>
                </form>

                {/* Participant list */}
                <div className="space-y-2 max-h-48 overflow-y-auto overflow-x-hidden pr-1">
                  {(contact.participants || []).map((participant) => {
                    const isAdmin = participant.role === "admin";
                    return (
                      <div
                        key={participant.id}
                        className="flex items-center justify-between p-1.5 rounded hover:bg-black/5 dark:hover:bg-white/5 text-xs"
                      >
                        <div className="flex items-center gap-2 min-w-0">
                          <div className="w-6 h-6 rounded-full bg-[#00a884]/20 text-[#00a884] flex items-center justify-center font-bold text-[10px]">
                            {participant.name.slice(0, 1).toUpperCase()}
                          </div>
                          <span className="font-bold truncate">{participant.name}</span>
                        </div>

                        <div className="flex items-center gap-1.5 shrink-0">
                          {isAdmin ? (
                            <span className="text-[8px] bg-emerald-500/10 text-emerald-500 px-1.5 py-0.5 rounded-full font-extrabold uppercase">
                              Admin
                            </span>
                          ) : (
                            <span className="text-[8px] opacity-40 uppercase">Member</span>
                          )}

                          {/* Controls (Disabled for self to prevent self-kicking) */}
                          {participant.id !== "me" && (
                            <div className="flex items-center gap-1">
                              <button
                                onClick={() => handleToggleAdmin(participant.id, participant.role)}
                                className={`p-1 rounded hover:bg-black/10 dark:hover:bg-white/10 ${
                                  isAdmin ? "text-orange-400" : "text-emerald-500"
                                }`}
                                title={isAdmin ? "Demote from Admin" : "Promote to Admin"}
                              >
                                {isAdmin ? <ShieldAlert size={10} /> : <ShieldCheck size={10} />}
                              </button>
                              <button
                                onClick={() => handleKickMember(participant.id)}
                                className="p-1 rounded hover:bg-black/10 dark:hover:bg-white/10 text-red-500"
                                title="Kick participant"
                              >
                                <Trash2 size={10} />
                              </button>
                            </div>
                          )}
                        </div>
                      </div>
                    );
                  })}
                </div>

                {/* Group policies settings */}
                <div className="space-y-2 pt-2 border-t border-black/5 dark:border-white/5">
                  <h4 className="text-[10px] font-bold uppercase opacity-55 tracking-wider">
                    Group Policy Settings
                  </h4>
                  <label className="flex items-center gap-2 text-xs cursor-pointer">
                    <input
                      type="checkbox"
                      checked={contact.onlyAdminsCanSend || false}
                      onChange={(e) => onUpdateContactSettings(contact.id, { onlyAdminsCanSend: e.target.checked })}
                      className="accent-[#00a884] rounded"
                    />
                    <span>Only Admins Can Send Messages</span>
                  </label>
                </div>
              </div>
            )}

            {/* 4. Chat/Contact general actions */}
            <div className="space-y-3 pt-3 border-t border-black/5 dark:border-white/5 text-xs">
              <h4 className="text-[10px] font-bold uppercase opacity-55 tracking-wider">
                Privacy & Action Center
              </h4>

              <div className="flex justify-between items-center p-1 rounded">
                <span className="font-semibold">Mute Chat Notifications</span>
                <input
                  type="checkbox"
                  checked={contact.isMuted || false}
                  onChange={(e) => onUpdateContactSettings(contact.id, { isMuted: e.target.checked })}
                  className="accent-[#00a884] w-4 h-4"
                />
              </div>

              {!contact.isGroup && (
                <div className="flex justify-between items-center p-1 rounded">
                  <span className="font-semibold text-red-500">Block Contact</span>
                  <input
                    type="checkbox"
                    checked={contact.isBlocked || false}
                    onChange={(e) => onUpdateContactSettings(contact.id, { isBlocked: e.target.checked })}
                    className="accent-red-500 w-4 h-4"
                  />
                </div>
              )}
            </div>
          </div>
        </div>
      )}

      {/* Lightbox / Zoom Image Modal */}
      {activeZoomImage && (
        <div className="fixed inset-0 bg-black/95 backdrop-blur-xs z-50 flex flex-col items-center justify-center p-4">
          <div className="absolute top-4 right-4 flex gap-4 z-50">
            <button
              onClick={(e) => {
                e.preventDefault();
                handleSafeDownload(activeZoomImage, "whatsapp_image.png");
              }}
              className="bg-white/10 hover:bg-white/20 text-white p-2.5 rounded-full shadow transition-all cursor-pointer"
              title="Download Image"
            >
              <Download size={20} />
            </button>
            <button
              onClick={() => setActiveZoomImage(null)}
              className="bg-white/10 hover:bg-white/20 text-white p-2.5 rounded-full shadow transition-all cursor-pointer"
            >
              <X size={20} />
            </button>
          </div>
          <img
            src={activeZoomImage || null}
            alt="Zoom Detail"
            className="max-h-[85vh] max-w-full object-contain rounded-lg shadow-2xl"
            style={{ animation: "scaleUp 0.15s ease-out" }}
            referrerPolicy="no-referrer"
          />
        </div>
      )}

      {/* Context Action Menu Overlay */}
      {contextMenuMsg && (() => {
        const menuWidth = 220;
        const menuHeight = 350;
        const screenWidth = window.innerWidth;
        const screenHeight = window.innerHeight;

        let left = contextMenuMsg.x;
        let top = contextMenuMsg.y;

        if (left + menuWidth > screenWidth) {
          left = screenWidth - menuWidth - 16;
        }
        if (top + menuHeight > screenHeight) {
          top = screenHeight - menuHeight - 16;
        }
        if (left < 16) left = 16;
        if (top < 16) top = 16;

        const msg = contextMenuMsg.msg;

        return (
          <div 
            className="fixed inset-0 z-50 overflow-hidden select-none"
            onClick={() => {
              setContextMenuMsg(null);
              setShowSubEmojiPicker(false);
            }}
            onContextMenu={(e) => {
              e.preventDefault();
              setContextMenuMsg(null);
              setShowSubEmojiPicker(false);
            }}
          >
            {/* Backdrop */}
            <div className="absolute inset-0 bg-black/10 dark:bg-black/25 transition-opacity" />

            {/* Menu Container */}
            <div
              className="absolute flex flex-col gap-1.5"
              style={{ 
                left: `${left}px`, 
                top: `${top}px`,
                animation: "scaleUp 0.1s ease-out" 
              }}
              onClick={(e) => e.stopPropagation()}
            >
              {/* Reaction Pill Indicator */}
              <div className="flex items-center gap-3.5 bg-white dark:bg-[#202c33] px-4 py-2 rounded-full shadow-lg border border-black/10 dark:border-white/10 w-fit">
                {["😮", "😢", "🙏"].map((emoji) => (
                  <button
                    key={emoji}
                    onClick={() => {
                      onAddReaction(msg.id, emoji);
                      setContextMenuMsg(null);
                      showToast(`Reacted with ${emoji}`);
                    }}
                    className="text-xl hover:scale-125 active:scale-95 transition-transform cursor-pointer select-none"
                  >
                    {emoji}
                  </button>
                ))}
                <button
                  onClick={() => {
                    setShowSubEmojiPicker(!showSubEmojiPicker);
                  }}
                  className="w-6 h-6 rounded-full flex items-center justify-center hover:bg-black/5 dark:hover:bg-white/10 text-gray-500 dark:text-gray-300 font-bold text-sm cursor-pointer"
                  title="More Reactions"
                >
                  +
                </button>
              </div>

              {/* Sub Emoji Expanded Picker Row */}
              {showSubEmojiPicker && (
                <div className="flex items-center gap-2 bg-white dark:bg-[#202c33] p-2 rounded-xl shadow-lg border border-black/10 dark:border-white/10 w-fit flex-wrap max-w-[220px]">
                  {["👍", "❤️", "😂", "🔥", "👏", "🎉", "💯"].map((emoji) => (
                    <button
                      key={emoji}
                      onClick={() => {
                        onAddReaction(msg.id, emoji);
                        setContextMenuMsg(null);
                        setShowSubEmojiPicker(false);
                        showToast(`Reacted with ${emoji}`);
                      }}
                      className="text-lg hover:scale-125 active:scale-95 transition-transform cursor-pointer"
                    >
                      {emoji}
                    </button>
                  ))}
                </div>
              )}

              {/* Main Action Menu Card */}
              <div className="bg-white dark:bg-[#202c33] rounded-2xl shadow-2xl border border-black/5 dark:border-white/5 py-1.5 w-52 overflow-hidden flex flex-col text-gray-800 dark:text-gray-200">
                <button
                  onClick={() => {
                    onSetQuotedMessage(msg);
                    setContextMenuMsg(null);
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-black/5 dark:hover:bg-white/5 text-left text-xs font-semibold w-full cursor-pointer transition-colors"
                >
                  <CornerUpLeft size={16} className="text-gray-500 dark:text-gray-400" />
                  <span>Reply</span>
                </button>

                <button
                  onClick={() => {
                    const textToCopy = msg.text || msg.fileName || "";
                    navigator.clipboard.writeText(textToCopy);
                    showToast("Message copied to clipboard (মেসেজ কপি করা হয়েছে)");
                    setContextMenuMsg(null);
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-black/5 dark:hover:bg-white/5 text-left text-xs font-semibold w-full cursor-pointer transition-colors"
                >
                  <Copy size={16} className="text-gray-500 dark:text-gray-400" />
                  <span>Copy</span>
                </button>

                <button
                  onClick={() => {
                    setShowSubEmojiPicker(!showSubEmojiPicker);
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-black/5 dark:hover:bg-white/5 text-left text-xs font-semibold w-full cursor-pointer transition-colors"
                >
                  <Smile size={16} className="text-gray-500 dark:text-gray-400" />
                  <span>React</span>
                </button>

                <button
                  onClick={() => {
                    setForwardMessageTarget(msg);
                    setIsForwardModalOpen(true);
                    setContextMenuMsg(null);
                    setForwardedStatus({});
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-black/5 dark:hover:bg-white/5 text-left text-xs font-semibold w-full cursor-pointer transition-colors"
                >
                  <CornerUpRight size={16} className="text-gray-500 dark:text-gray-400" />
                  <span>Forward</span>
                </button>

                <button
                  onClick={() => {
                    onPinMessage(msg.id);
                    setContextMenuMsg(null);
                    showToast(msg.isPinned ? "Message unpinned" : "Message pinned to top of conversation");
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-black/5 dark:hover:bg-white/5 text-left text-xs font-semibold w-full cursor-pointer transition-colors"
                >
                  <svg className={`w-4 h-4 text-gray-500 dark:text-gray-400 ${msg.isPinned ? "fill-current text-emerald-500" : ""}`} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                    <line x1="12" y1="17" x2="12" y2="22" />
                    <path d="M5 17h14v-1.76a2 2 0 0 0-.44-1.24l-2.12-2.6A2 2 0 0 1 16 10.16V6a2 2 0 0 0-2-2h-4a2 2 0 0 0-2 2v4.16a2 2 0 0 1-.44 1.24l-2.12 2.6A2 2 0 0 0 5 15.24z" />
                  </svg>
                  <span>{msg.isPinned ? "Unpin" : "Pin"}</span>
                </button>

                <button
                  onClick={() => {
                    onStarMessage(msg.id);
                    setContextMenuMsg(null);
                    showToast(msg.isStarred ? "Message unstarred" : "Message starred (স্টার মার্ক করা হয়েছে)");
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-black/5 dark:hover:bg-white/5 text-left text-xs font-semibold w-full cursor-pointer transition-colors"
                >
                  <svg className={`w-4 h-4 text-gray-500 dark:text-gray-400 ${msg.isStarred ? "fill-amber-400 text-amber-400" : ""}`} viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2">
                    <polygon points="12 2 15.09 8.26 22 9.27 17 14.14 18.18 21.02 12 17.77 5.82 21.02 7 14.14 2 9.27 8.91 8.26 12 2" />
                  </svg>
                  <span>{msg.isStarred ? "Unstar" : "Star"}</span>
                </button>

                <div className="border-t border-black/5 dark:border-white/5 my-1" />

                <button
                  onClick={() => {
                    onDeleteMessage(msg.id);
                    setContextMenuMsg(null);
                    showToast("Message deleted successfully (মেসেজ ডিলিট করা হয়েছে)");
                  }}
                  className="flex items-center gap-3 px-4 py-2 hover:bg-red-500/10 hover:text-red-500 text-left text-xs font-semibold w-full text-red-500 cursor-pointer transition-colors"
                >
                  <Trash2 size={16} className="text-red-500" />
                  <span>Delete</span>
                </button>
              </div>
            </div>
          </div>
        );
      })()}

      {/* Forward Message Modal */}
      {isForwardModalOpen && forwardMessageTarget && (
        <div className="fixed inset-0 z-50 flex items-center justify-center p-4 select-none">
          {/* Backdrop */}
          <div 
            className="absolute inset-0 bg-black/60 backdrop-blur-xs"
            onClick={() => {
              setIsForwardModalOpen(false);
              setForwardMessageTarget(null);
            }}
          />

          {/* Modal Container */}
          <div 
            className={`relative w-full max-w-md rounded-2xl shadow-2xl border flex flex-col max-h-[500px] overflow-hidden ${
              isDark ? "bg-[#222e35] border-gray-800 text-white" : "bg-white border-gray-100 text-gray-800"
            }`}
            style={{ animation: "scaleUp 0.15s ease-out" }}
            onClick={(e) => e.stopPropagation()}
          >
            {/* Header */}
            <div className="flex items-center justify-between px-5 py-4 border-b border-black/5 dark:border-white/5">
              <h3 className="font-bold text-sm flex items-center gap-2">
                <svg className="w-4 h-4 text-emerald-500" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2.5">
                  <polyline points="17 1 21 5 17 9" />
                  <path d="M3 11V9a4 4 0 0 1 4-4h14" />
                </svg>
                Forward Message (মেসেজ ফরওয়ার্ড করুন)
              </h3>
              <button
                onClick={() => {
                  setIsForwardModalOpen(false);
                  setForwardMessageTarget(null);
                }}
                className="p-1.5 rounded-full hover:bg-black/5 dark:hover:bg-white/10 text-gray-500 dark:text-gray-400 cursor-pointer"
              >
                <X size={18} />
              </button>
            </div>

            {/* Message Preview Banner */}
            <div className="bg-black/5 dark:bg-white/5 px-5 py-2.5 text-xs text-gray-500 dark:text-gray-400 italic border-b border-black/5 dark:border-white/5 truncate">
              Content: "{forwardMessageTarget.text || `${forwardMessageTarget.type} file`}"
            </div>

            {/* Search Input */}
            <div className="p-3 border-b border-black/5 dark:border-white/5">
              <div className="relative">
                <Search size={14} className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-400" />
                <input
                  type="text"
                  placeholder="Search contacts..."
                  value={forwardSearchQuery}
                  onChange={(e) => setForwardSearchQuery(e.target.value)}
                  className={`w-full pl-9 pr-4 py-2 text-xs rounded-lg border outline-none transition-colors ${
                    isDark 
                      ? "bg-[#2a3942] border-gray-700 text-white focus:border-emerald-500" 
                      : "bg-gray-50 border-gray-200 text-gray-800 focus:border-emerald-500"
                  }`}
                />
              </div>
            </div>

            {/* Contacts List */}
            <div className="flex-1 overflow-y-auto overflow-x-hidden divide-y divide-black/5 dark:divide-white/5 p-2">
              {(() => {
                const filtered = contacts.filter((c) => 
                  c.name.toLowerCase().includes(forwardSearchQuery.toLowerCase()) ||
                  (c.phoneNumber && c.phoneNumber.includes(forwardSearchQuery))
                );

                if (filtered.length === 0) {
                  return (
                    <div className="py-8 text-center text-xs opacity-50 italic">
                      No matching contacts found.
                    </div>
                  );
                }

                return filtered.map((c) => {
                  const hasForwarded = forwardedStatus[c.id];
                  return (
                    <div key={c.id} className="flex items-center justify-between py-2 px-3 hover:bg-black/5 dark:hover:bg-white/5 rounded-lg transition-colors">
                      <div className="flex items-center gap-3">
                        <div className="w-9 h-9 rounded-full bg-emerald-500/10 flex items-center justify-center overflow-hidden shrink-0">
                          {c.avatar && c.avatar.startsWith("http") ? (
                            <img src={c.avatar} alt={c.name} className="w-full h-full object-cover" referrerPolicy="no-referrer" />
                          ) : (
                            <span className="text-emerald-600 font-semibold text-xs">
                              {c.name.slice(0, 2).toUpperCase()}
                            </span>
                          )}
                        </div>
                        <div className="min-w-0">
                          <h4 className="font-semibold text-xs truncate max-w-[180px]">{c.name}</h4>
                          <p className="text-[10px] opacity-60 truncate">
                            {c.isGroup ? "Group Chat" : c.phoneNumber || "WhatsApp contact"}
                          </p>
                        </div>
                      </div>

                      <button
                        onClick={() => {
                          if (hasForwarded) return;
                          onForwardMessage(forwardMessageTarget, c.id);
                          setForwardedStatus((prev) => ({ ...prev, [c.id]: true }));
                          showToast(`Forwarded to ${c.name}`);
                        }}
                        disabled={hasForwarded}
                        className={`px-3 py-1.5 rounded-full font-semibold text-[10px] shadow-xs cursor-pointer transition-all ${
                          hasForwarded
                            ? "bg-emerald-500/10 text-emerald-500 border border-emerald-500/20"
                            : "bg-[#00a884] hover:bg-[#008069] text-white"
                        }`}
                      >
                        {hasForwarded ? "✓ Sent" : "Forward"}
                      </button>
                    </div>
                  );
                });
              })()}
            </div>

            {/* Footer */}
            <div className="p-3 border-t border-black/5 dark:border-white/5 flex justify-end">
              <button
                onClick={() => {
                  setIsForwardModalOpen(false);
                  setForwardMessageTarget(null);
                }}
                className="px-4 py-2 bg-[#00a884] hover:bg-[#008069] text-white text-xs font-bold rounded-lg transition-colors shadow-sm cursor-pointer"
              >
                Done
              </button>
            </div>
          </div>
        </div>
      )}

      {/* Slide-in Toast Notification Card */}
      {toastMessage && (
        <div 
          className="fixed bottom-6 left-1/2 -translate-x-1/2 bg-[#0b141a]/90 text-white text-xs font-semibold px-5 py-3 rounded-xl shadow-2xl z-50 flex items-center gap-2 border border-white/10"
          style={{ 
            animation: "slideInUp 0.18s cubic-bezier(0.16, 1, 0.3, 1)" 
          }}
        >
          <svg className="w-4 h-4 text-emerald-500 shrink-0" fill="none" stroke="currentColor" strokeWidth="3" viewBox="0 0 24 24">
            <polyline points="20 6 9 17 4 12" />
          </svg>
          <span>{toastMessage}</span>
        </div>
      )}
    </div>
  );
};
